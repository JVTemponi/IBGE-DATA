import csv
import requests
import json
import argparse
import sys
import math

# --- CONFIGURAÇÕES ---
ONE_SIGNAL_APP_ID = "SEU_APP_ID_AQUI"
ONE_SIGNAL_API_KEY = "SUA_API_KEY_AQUI"

# Limite de IDs por requisição (O OneSignal recomenda max 2000 para player_ids)
TAMANHO_DO_LOTE = 1000 

def configurar_argumentos():
    """Configura os argumentos da linha de comando."""
    parser = argparse.ArgumentParser(description='Dispara Push via OneSignal baseado em CSV.')
    parser.add_argument('arquivo_csv', type=str, help='Caminho do CSV com player_ids')
    parser.add_argument('--titulo', type=str, default='Novidade!', help='Título da notificação')
    parser.add_argument('--mensagem', type=str, required=True, help='Corpo da mensagem')
    parser.add_argument('--url', type=str, default='', help='Link para abrir ao clicar (opcional)')
    return parser.parse_args()

def ler_ids_usuarios(caminho_csv):
    """Lê o CSV e busca a coluna 'player_id'."""
    ids = []
    try:
        with open(caminho_csv, mode='r', encoding='utf-8') as arquivo:
            leitor = csv.DictReader(arquivo)
            
            # Verifica se a coluna correta existe
            if 'player_id' not in leitor.fieldnames:
                # Tenta verificar se existe 'id' ou 'user_id' como fallback
                if 'id' in leitor.fieldnames:
                    coluna_alvo = 'id'
                else:
                    print("ERRO: O CSV precisa ter uma coluna chamada 'player_id'.")
                    sys.exit(1)
            else:
                coluna_alvo = 'player_id'

            for linha in leitor:
                uid = linha[coluna_alvo].strip()
                if uid:
                    ids.append(uid)
        return ids
    except FileNotFoundError:
        print(f"ERRO: Arquivo '{caminho_csv}' não encontrado.")
        sys.exit(1)

def enviar_batch_push(lista_ids, titulo, mensagem, url):
    """Envia o lote para a API."""
    header = {
        "Content-Type": "application/json; charset=utf-8",
        "Authorization": f"Basic {ONE_SIGNAL_API_KEY}"
    }

    payload = {
        "app_id": ONE_SIGNAL_APP_ID,
        "headings": {"en": titulo},
        "contents": {"en": mensagem},
        "include_player_ids": lista_ids,
        # Ícone pequeno (opcional, usa o padrão do app se omitido)
        # "small_icon": "ic_stat_onesignal_default", 
    }

    # Adiciona URL apenas se foi passada
    if url:
        payload["url"] = url

    try:
        req = requests.post(
            "https://onesignal.com/api/v1/notifications",
            headers=header,
            data=json.dumps(payload)
        )
        
        if req.status_code == 200:
            print(f"-> Lote enviado. Status: {req.json().get('id', 'OK')}")
            # Verifica se houve erros com IDs específicos (ex: tokens inválidos/desinscritos)
            recipients = req.json().get('recipients', 0)
            print(f"   Destinatários válidos neste lote: {recipients}")
        else:
            print(f"-> Erro API: {req.status_code} - {req.text}")

    except Exception as e:
        print(f"-> Falha de conexão: {e}")

def processar_envio(ids, args):
    """Gerencia a quebra em lotes."""
    total = len(ids)
    if total == 0:
        print("Nenhum ID encontrado no CSV.")
        return

    qtd_lotes = math.ceil(total / TAMANHO_DO_LOTE)
    print(f"Iniciando Push para {total} dispositivos em {qtd_lotes} lotes.")

    for i in range(0, total, TAMANHO_DO_LOTE):
        lote = ids[i:i + TAMANHO_DO_LOTE]
        enviar_batch_push(lote, args.titulo, args.mensagem, args.url)

# --- EXECUÇÃO ---
if __name__ == "__main__":
    args = configurar_argumentos()
    
    print(f"Lendo base: {args.arquivo_csv}")
    lista_ids = ler_ids_usuarios(args.arquivo_csv)
    
    processar_envio(lista_ids, args)
