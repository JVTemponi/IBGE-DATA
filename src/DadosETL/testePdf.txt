import fitz  # PyMuPDF
import pandas as pd
import re

def extrair_dados_folha(caminho_pdf):
    """
    Função principal para extrair dados de uma folha de pagamento em PDF,
    processar as informações de cada funcionário e retornar um DataFrame do pandas.
    """
    # Abre o arquivo PDF
    doc = fitz.open(caminho_pdf)
    texto_completo = ""
    # Concatena o texto de todas as páginas
    for pagina in doc:
        texto_completo += pagina.get_text("text")
    doc.close()

    # Divide o texto completo em blocos por funcionário usando o separador
    blocos_funcionarios = texto_completo.split('--------------------------------------------------------------------------------')

    dados_todos_funcionarios = []
    colunas_proventos_descontos = set()

    # Primeiro loop: identificar todas as colunas de proventos e descontos
    for bloco in blocos_funcionarios:
        if not bloco.strip():
            continue
        try:
            # Extrai apenas a parte relevante do bloco (até a linha do líquido)
            bloco_relevante = bloco.split('LÍQUIDO:')[0]

            # Encontra todas as linhas que parecem ser de proventos ou descontos
            linhas_eventos = re.findall(r'^\s*(\d+)\s+([A-ZÀ-Ú\s\d\.\/\%]+?)\s+', bloco_relevante, re.MULTILINE)
            for codigo, descricao in linhas_eventos:
                colunas_proventos_descontos.add(f"{codigo.strip()} - {descricao.strip()}")
        except Exception:
            # Ignora blocos que não podem ser processados
            continue

    colunas_ordenadas_eventos = sorted(list(colunas_proventos_descontos))

    # Segundo loop: extrair os dados de cada funcionário
    for bloco in blocos_funcionarios:
        if not bloco.strip():
            continue

        try:
            dados_funcionario = {}
            # Extrai a parte relevante do bloco
            bloco_relevante, parte_liquido = bloco.split('LÍQUIDO:')
            bloco_relevante += 'LÍQUIDO:' # Adiciona a linha do líquido de volta para extração

            # Extração de dados fixos usando expressões regulares (ajuste se necessário)
            matricula = re.search(r'(\d+-\d+/\d+)', bloco)
            dados_funcionario['Matrícula'] = matricula.group(1) if matricula else None

            nome = re.search(r'Nome:\s*([A-Z\s]+)', bloco)
            dados_funcionario['Nome'] = nome.group(1).strip() if nome else None
            
            cargo = re.search(r'Cargo:\s*([A-Z\s]+)', bloco)
            dados_funcionario['Cargo'] = cargo.group(1).strip() if cargo else None

            # Extração dos totais
            proventos = re.search(r'PROVENTOS:\s*([\d\.,]+)', bloco_relevante)
            dados_funcionario['Total Proventos'] = proventos.group(1) if proventos else None

            descontos = re.search(r'DESCONTOS:\s*([\d\.,]+)', bloco_relevante)
            dados_funcionario['Total Descontos'] = descontos.group(1) if descontos else None

            liquido = re.search(r'LÍQUIDO:\s*([\d\.,]+)', parte_liquido)
            dados_funcionario['Líquido'] = liquido.group(1).strip().split('\n')[0] if liquido else None


            # Inicializa as colunas de eventos com vazio
            for col in colunas_ordenadas_eventos:
                dados_funcionario[col] = ''

            # Extrai os valores de cada evento de provento/desconto
            linhas_bloco = bloco_relevante.split('\n')
            for linha in linhas_bloco:
                match = re.match(r'^\s*(\d+)\s+([A-ZÀ-Ú\s\d\.\/\%]+?)\s+.*?([\d\.,]+)\s*$', linha)
                if match:
                    codigo = match.group(1).strip()
                    descricao = match.group(2).strip()
                    valor = match.group(3).strip()
                    nome_coluna = f"{codigo} - {descricao}"
                    if nome_coluna in dados_funcionario:
                        dados_funcionario[nome_coluna] = valor

            dados_todos_funcionarios.append(dados_funcionario)

        except Exception as e:
            print(f"Aviso: Não foi possível processar um bloco de funcionário. Erro: {e}")
            continue

    # Cria o DataFrame final
    df = pd.DataFrame(dados_todos_funcionarios)
    
    # Reordena as colunas para uma melhor visualização
    colunas_fixas = ['Matrícula', 'Nome', 'Cargo', 'Total Proventos', 'Total Descontos', 'Líquido']
    colunas_finais = colunas_fixas + colunas_ordenadas_eventos
    df = df[colunas_finais]

    return df

# --- Execução do Script ---
if __name__ == "__main__":
    # Substitua 'sua_folha_de_pagamento.pdf' pelo nome do seu arquivo PDF
    caminho_do_pdf = 'sua_folha_de_pagamento.pdf'
    
    try:
        df_resultado = extrair_dados_folha(caminho_do_pdf)
        
        # Salva o resultado em um arquivo CSV
        # O encoding 'utf-8-sig' é recomendado para compatibilidade com o Excel
        df_resultado.to_csv('folha_de_pagamento_extraida.csv', index=False, sep=';', encoding='utf-8-sig')
        
        print("Extração concluída com sucesso!")
        print(f"O arquivo 'folha_de_pagamento_extraida.csv' foi gerado.")
        
    except FileNotFoundError:
        print(f"Erro: O arquivo '{caminho_do_pdf}' não foi encontrado.")
    except Exception as e:
        print(f"Ocorreu um erro inesperado: {e}")
